01介绍

中午访问自己的博客 发现不可访问，ping ip也不行，找到阿里云服务器，咨询了下 路由有问题，最后发现ip不能国内访问了，博主用的国内域名、境外服务器省去备案. 果断选择退款了，改用腾讯轻量云，韩国节点，以后就按月续费了.

这次记录一遍重新部署流程 方便以后查看

02重新部署

系统选择

Ubuntu Server 24.04 LTS 64bit

之前用的centos 腾讯云要升级8以上用自研的系统 还是选择Ubuntu，体验下

注意登录后设置root

sudo passwd root

sudo vi /etc/ssh/sshd_config
#Authentication，将 PermitRootLogin 参数修改为 yes。如果 PermitRootLogin 参数被注释，请去掉首行的注释符号（#）,将 PasswordAuthentication 参数修改为 yes

安装

sudo apt update

sudo apt install nginx

sudo nginx -s reload

vim /etc/nginx/nginx.conf

sudo nginx -t #测试配置是否正常


# docker

sudo apt install apt-transport-https ca-certificates curl software-properties-common

curl -fsSL https://download.docker.com/linux/ubuntu/gpg | sudo apt-key add -

sudo add-apt-repository "deb [arch=amd64] https://download.docker.com/linux/ubuntu $(lsb_release -cs) stable"

sudo apt update

sudo apt install docker-ce

nginx配置

在http模块配置

            server {
                listen 80;
                server_name hanami.top www.hanami.top;  # 监听80端口，并设置域名和www子域名

                    # 强制HTTP请求重定向到HTTPS
                    return 301 https://$server_name$request_uri;
            }

            server {
                listen       443 ssl http2;
                listen       [::]:443 ssl http2;
                server_name hanami.top;

                ssl_certificate "/etc/nginx/cert/hanami.top.pem";
                ssl_certificate_key "/etc/nginx/cert/hanami.top.key";
                ssl_session_cache shared:SSL:1m;
                ssl_session_timeout  10m;
                ssl_ciphers ECDHE-RSA-AES128-GCM-SHA256:ECDHE:ECDH:AES:HIGH:!NULL:!aNULL:!MD5:!ADH:!RC4;
                ssl_prefer_server_ciphers on;

                # Load configuration files for the default server block.
                include /etc/nginx/default.d/*.conf;

                location /blog {
                        proxy_pass http://localhost:3435;
                         proxy_set_header Host $host;
                         proxy_set_header X-Real-IP $remote_addr;
                         proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
                         proxy_set_header X-Forwarded-Proto $scheme;
                 }

                error_page 404 /404.html;

                }

                }
            }

配置自动化

tx控制台添加：link

利用github的action推送这个私有镜像即可

name: Docker Image CI

on:
  push: # push 时触发ci
    branches: [main] # 作用于main分支
  # pull_request:
  #   branches: [main]

jobs:
  build:
    runs-on: ubuntu-latest

    steps:
      # 拉取main分支代码
      - name: Checkout
        uses: actions/checkout@v2

      # 制作docker镜像并推送到阿里云容器镜像服务
      - name: build and push docker image
        run: |
          echo ${{ secrets.ALIYUNIMAGE_PASSWORD }} | docker login krccr.ccs.tencentyun.com --username ${{ secrets.ALIYUNIMAGE_USERNAME }} --password-stdin

          docker image build -t blog:latest .
          docker tag blog krccr.ccs.tencentyun.com/hanamitop/blog:latest
          docker push krccr.ccs.tencentyun.com/hanamitop/blog:latest
          docker logout
      # 登录远程服务器，拉取镜像，制作并重启容器
      # https://github.com/marketplace/actions/remote-ssh-commands
      - name: ssh remote deploy
        uses: fifsky/ssh-action@master
        with:
          command: |
            cd /
            echo -e "1.docker login start==>"
            echo ${{ secrets.ALIYUNIMAGE_PASSWORD }} | docker login krccr.ccs.tencentyun.com --username ${{ secrets.ALIYUNIMAGE_USERNAME }} --password-stdin

            echo -e "2.docker stop blog container==>"
            docker container stop blog

            echo -e "3.docker conatainer rm==>"
            docker container rm blog

            echo -e "4.docker image rm==>"
            docker image rm krccr.ccs.tencentyun.com/hanamitop/blog:latest

            echo -e "5.docker pull==>"
            docker pull krccr.ccs.tencentyun.com/hanamitop/blog:latest

            echo -e "6.docker container create and start==>"
            docker container run -d -p 3435:3435 --name blog krccr.ccs.tencentyun.com/hanamitop/blog:latest

            echo -e "7.docker logout==>"
            docker logout
          host: ${{ secrets.HOST }}
          user: ${{ secrets.USER }}
          pass: ${{ secrets.ALIYUNCN_PASSWORD }}
